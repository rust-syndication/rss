166impl FromXml for Person {
167    fn from_xml<B: BufRead>(reader: &mut Reader<B>, _: Attributes<'_>) -> Result<Self, Error> {
168        let mut person = Person::default();
169        let mut buf = Vec::new();
170
171        loop {
172            match reader.read_event_into(&mut buf).map_err(XmlError::new)? {
173                Event::Start(element) => match decode(element.name().as_ref(), reader)? {
174                    Cow::Borrowed("name") => person.name = atom_text(reader)?.unwrap_or_default(),
175                    Cow::Borrowed("email") => person.email = atom_text(reader)?,
176                    Cow::Borrowed("uri") => person.uri = atom_text(reader)?,
177                    n => {
178                        if let Some((ns, name)) = extension_name(n.as_ref()) {
179                            parse_extension(
180                                reader,
181                                element.attributes(),
182                                ns,
183                                name,
184                                &mut person.extensions,
185                            )?;
186                        } else {
187                            skip(element.name(), reader)?;
188                        }
189                    }
190                },
191                Event::End(_) => break,
192                Event::Eof => return Err(Error::Eof),
193                _ => {}
194            }
195
196            buf.clear();
197        }
198
199        Ok(person)
200    }
201}

203impl ToXmlNamed for Person {
204    fn to_xml_named<W>(&self, writer: &mut Writer<W>, name: &str) -> Result<(), XmlError>
205    where
206        W: Write,
207    {
208        writer
209            .write_event(Event::Start(BytesStart::new(name)))
210            .map_err(XmlError::new)?;
211        writer.write_text_element("name", &self.name)?;
212
213        if let Some(ref email) = self.email {
214            writer.write_text_element("email", email)?;
215        }
216
217        if let Some(ref uri) = self.uri {
218            writer.write_text_element("uri", uri)?;
219        }
220
221        for map in self.extensions.values() {
222            for extensions in map.values() {
223                writer.write_objects(extensions)?;
224            }
225        }
226
227        writer
228            .write_event(Event::End(BytesEnd::new(name)))
229            .map_err(XmlError::new)?;
230
231        Ok(())
232    }
233}
