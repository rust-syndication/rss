19pub(crate) fn decode<'s, B: BufRead>(
20    bytes: &'s [u8],
21    reader: &Reader<B>,
22) -> Result<Cow<'s, str>, Error> {
23    let text = reader.decoder().decode(bytes)?;
24    Ok(text)
25}
26
27pub(crate) fn attr_value<'s, B: BufRead>(
28    attr: &'s Attribute<'s>,
29    reader: &Reader<B>,
30) -> Result<Cow<'s, str>, Error> {
31    let value = attr.decode_and_unescape_value(reader.decoder())?;
32    Ok(value)
33}
34
35pub(crate) fn skip<B: BufRead>(end: QName<'_>, reader: &mut Reader<B>) -> Result<(), Error> {
36    reader.read_to_end_into(end, &mut Vec::new())?;
37    Ok(())
38}

40pub fn element_text<R: BufRead>(reader: &mut Reader<R>) -> Result<Option<String>, Error> {
41    let mut content = String::new();
42    let mut buf = Vec::new();
43
44    loop {
45        match reader.read_event_into(&mut buf)? {
46            Event::Start(element) => {
47                skip(element.name(), reader)?;
48            }
49            Event::Text(element) => {
50                let decoded = element.decode()?;
51                content.push_str(decoded.as_ref());
52            }
53            Event::GeneralRef(gref) => {
54                let entity = gref.decode()?;
55                if let Some(resolved_entity) = resolve_predefined_entity(&entity) {
56                    content.push_str(resolved_entity);
57                } else if let Some(ch) = gref.resolve_char_ref()? {
58                    content.push(ch);
59                } else {
60                    content.push('&');
61                    content.push_str(&entity);
62                    content.push(';');
63                }
64            }
65            Event::CData(element) => {
66                content.push_str(decode(&element, reader)?.as_ref());
67            }
68            Event::End(_) | Event::Eof => break,
69            _ => {}
70        }
71        buf.clear();
72    }
73
74    Ok(Some(content.trim().to_owned()).filter(|c| !c.is_empty()))
75}
